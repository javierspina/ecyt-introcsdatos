  filter(escucha_musica=='SI' & miro_tv=='SI' & NSEcat1 != '') %>% 
  group_by(NSEcat1, concurre_teatro) %>% 
  mutate(cantidad = n()) %>% 
  ggplot() +
  geom_col(aes(x=cantidad, y=NSEcat1, fill=concurre_teatro), position='fill') +
